跨域配置

# 跨域问题

后端接口 127.0.0.1:8080/api/...

# 开发环境

// axios.js
const BASE_URL = '/api'

// vue.config.js
devServer: {
    host: '127.0.0.1',
    port: 8081,  // 前端运行的端口
    proxy: {
      '/': {
        ws: false,
        target: 'http://127.0.0.1:8080',  // 后端api
        changeOrigin: true,
        pathRewrite: {
          '^/': ''
        }
      }
    }
  },
