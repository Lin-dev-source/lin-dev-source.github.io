# ScrollReveal

# DEFAULT VALUE

{
    delay: 0, // 延迟时间
    distance: '0px', // 滚动的距离，单位可以用%，rem等
    duration: 600, // 动画的时长
    easing: 'cubic-bezier(0.5, 0, 0, 1)', // 动画属性
    interval: 0, // options.interval is the time between each reveal.
    opacity: 0,  // 动画开始时的透明度
    origin: 'bottom',  // 动画开始的位置，'bottom', 'left', 'top', 'right'
    rotate: {  //  动画开始前的旋转角度
        x: 0,
        y: 0,
        z: 0,
    },
    scale: 1,  // 动画开始前的伸缩
    cleanup: false, // 动画结束后调用clean，但如果reset为true就永远不会结束
    container: document.documentElement, // 监听滚动的元素
    desktop: true,
    mobile: true, // 在移动端是否使用动画
    reset: false,  // 回滚的时候是否再次触发动画
    useDelay: 'always',
            // 'always'	Delay all reveal animations.
            //'once'	Only delay the first time reveals occur.
            //'onload'	Only delay reveals triggered on page load.
    viewFactor: 0.0, // 元素进入窗口多少开始动画
    viewOffset: { //在计算元素可见性时扩展/收缩视口的活动边界。
        top: 0,
        right: 0,
        bottom: 0,
        left: 0,
    },
    afterReset: function (el) {}, // 当reset为true时，reset后会回调此函数
    afterReveal: function (el) {},// 动画结束后回调
    beforeReset: function (el) {},
    beforeReveal: function (el) {},
}
